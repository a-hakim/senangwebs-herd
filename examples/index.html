Quick Start

Installation

Include the CSS and JavaScript files in your HTML: 

<link rel="stylesheet" href="path/to/swh.css">
<script src="path/to/swh.js"></script>

HTML Initialization 

<div data-swh
     data-swh-storage-key="my-tabs"
     data-swh-default-tab="home"
     data-swh-allow-close>
    
    <div data-swh-tabs>
        <button data-swh-tab="home">Home</button>
        <button data-swh-tab="about">About</button>
    </div>
    
    <div data-swh-content>
        <iframe data-swh-iframe="home" data-url="home.html"></iframe>
        <iframe data-swh-iframe="about" data-url="about.html"></iframe>
    </div>
</div>

JavaScript Initialization 

const swh = new SWH({
    container: document.querySelector('#swh-container'),
    tabsContainer: document.querySelector('#swh-tabs'),
    contentContainer: document.querySelector('#swh-content'),
    presetTabs: [
        { id: "home", title: "Home", url: "home.html" }
    ],
    storageKey: "my-tabs",
    defaultTab: "home",
    allowClose: true,
    maxTabs: 5
});

API Methods 

// Open a new tab
swh.openTab('settings', 'Settings', 'settings.html');

// Switch to a tab
swh.switchTab('home');

// Close a tab
swh.closeTab('settings');

// Get open tabs
const tabs = swh.getOpenTabs();

// Listen to events
swh.on('tabOpened', (data) => {
    console.log('Tab opened:', data);
});

Running Examples Locally

To avoid CORS issues with iframes, run a local web server: 

# Using Python
python -m http.server 8000

# Using PHP
php -S localhost:8000

# Using Node.js
npx http-server -p 8000

Then visit: http://localhost:8000/examples/
