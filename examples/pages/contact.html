SWH API Methods

// Open a new tab
swh.openTab('tab-id', 'Tab Title', 'page.html');

// Switch to a tab
swh.switchTab('tab-id');

// Close a tab
swh.closeTab('tab-id');

// Get open tabs
const tabs = swh.getOpenTabs();

// Get active tab
const activeId = swh.getActiveTab();

// Reload a tab
swh.reloadTab('tab-id');

// Listen to events
swh.on('tabOpened', (data) => {
    console.log('Tab opened:', data);
});
